プログラム
int m=20;
int[]x=new int[m];
int[]y=new int[m];
int i=0;
void setup(){
size(300,300);
background(255);
}

void draw(){
if(keyPressed==true){
if(key=='p'){
print("始点を記録します ");
x[i]=mouseX;
y[i]=mouseY;
println(x[i],y[i]);
i=i+1;
delay(1500);
}
else if(key=='q'){
print("終点を記録します ");
x[i]=mouseX;
y[i]=mouseY;
println(x[i],y[i]);
i=i+1;
delay(1500);
}
else if(key=='g'){
print(i);
print("線を描きます ");
println("座標は",x[i-2],y[i-2],x[i-1],y[i-1],"です");
line(x[i-2],y[i-2],x[i-1],y[i-1]);
delay(1000);
}
else if(key=='f'){
for(i=0;i<=m;i=i+2){
println("line(",x[i],y[i],x[i+1],y[i+1],")");
delay(500);
}
}
}
}
使い方
mに画数を代入する
マウスを始める位置に持っていき、キーボードのPを押す
マウスを終わりの位置に持っていき、キーボードのQを押す
PとQを一回ずつ押したら、Gを押すと線が描ける
最後にキーボードのFを押すと線の座標がコンソールに表示される
